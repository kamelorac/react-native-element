const { height } = Dimensions.get('window');

  const [open, setOpen] = React.useState(false);

      visible={open}
      onOpen={() => {
        setOpen(true);
      }}
      onClose={() => {
        setOpen(false);

  );
};
const TooltipComponent = () => {
